404/********************************************************************************************************
405 * sl_status_get_string_n()
406 *
407 * @brief Get a copy of the status string associated to the status code passed, up to
408 * 'buffer_length' length, if the string associated to the status code is enabled. If not,
409 * the error code number, in hex, prefixed by "SL_STATUS_" will be copied in the buffer
410 * instead.
411 * For example, the buffer would either contain "SL_STATUS_FAIL" if that status string is
412 * enabled, or "SL_STATUS_0x0001" if the string is disabled, as SL_STATUS_FAIL's
413 * value is 0x0001.
414 *
415 * @param status The status code from which to obtain the status string.
416 *
417 * @param buffer Pointer to a buffer in which the status string will be copied. A terminating
418 * null-character will be appended after the copied status string.
419 *
420 * @param buffer_length Maximum number of characters that can be written in the buffer, including the
421 * terminating null-character. If the status string would be longer than the
422 * available length, it will be truncated and a null-terminating character will
423 * be the last character contained in the buffer.
424 *
425 * @return The number of characters that would have been written if the buffer_length had been
426 * sufficiently large, not counting the terminating null character.
427 * If the status code is invalid, 0 or a negative number is returned.
428 * Notice that only when this returned value is strictly positive and less than
429 * buffer_length, the status string has been completely written in the buffer.
430 *******************************************************************************************************/
431int32_t sl_status_get_string_n(sl_status_t status, char *buffer, uint32_t buffer_length);
