313__STATIC_INLINE uint32_t SL_RBIT(uint32_t value)
314{
315 uint32_t result;
316
317#if defined(__CORTEX_M) && (__CORTEX_M >= 0x03U)
318 result = __RBIT(value);
319#else
320 int32_t s = 4 * 8 - 1;
321
322 result = value;
323 for (value >>= 1U; value != 0U; value >>= 1U) {
324 result <<= 1U;
325 result |= value & 1U;
326 s--;
327 }
328 result <<= s;
329#endif
330 return result;
331}
